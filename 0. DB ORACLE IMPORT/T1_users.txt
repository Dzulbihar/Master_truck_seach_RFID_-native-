ALTER TABLE DASHBOARD.V_USERS
 DROP PRIMARY KEY CASCADE;

DROP TABLE DASHBOARD.T1_USERS CASCADE CONSTRAINTS;

CREATE TABLE DASHBOARD.T1_USERS
(
  ID        NUMBER,
  NAME      VARCHAR2(20 BYTE),
  USERNAME  VARCHAR2(50 BYTE),
  EMAIL     VARCHAR2(50 BYTE),
  PASSWORD  VARCHAR2(50 BYTE)
)
TABLESPACE USERS
PCTUSED    0
PCTFREE    10
INITRANS   1
MAXTRANS   255
STORAGE    (
            INITIAL          64K
            NEXT             1M
            MINEXTENTS       1
            MAXEXTENTS       UNLIMITED
            PCTINCREASE      0
            BUFFER_POOL      DEFAULT
           )
LOGGING 
NOCOMPRESS 
NOCACHE
NOPARALLEL
MONITORING;


CREATE OR REPLACE TRIGGER DASHBOARD.USERS_seq 
BEFORE INSERT ON DASHBOARD.V_USERS 
FOR EACH ROW
BEGIN
  SELECT USERS_seq.NEXTVAL
  INTO   :new.id
  FROM   dual;
END;
/


ALTER TABLE DASHBOARD.V_USERS ADD (
  PRIMARY KEY
  (ID)
  USING INDEX
    TABLESPACE USERS
    PCTFREE    10
    INITRANS   2
    MAXTRANS   255
    STORAGE    (
                INITIAL          64K
                NEXT             1M
                MINEXTENTS       1
                MAXEXTENTS       UNLIMITED
                PCTINCREASE      0
               ));
